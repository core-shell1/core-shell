## 💳 크레딧 시스템 DB 스키마 (원장 기반)

### PostgreSQL 테이블 설계 (Prisma ORM)

```prisma
// schema.prisma
model User {
  id            String   @id @default(cuid())
  email         String   @unique
  name          String?
  role          Role     @default(GENERAL) // GENERAL/PRO/ENTERPRISE
  creditBalance Int      @default(3)       // 실시간 잔액 (캐시)
  
  credits       Credit[]        // 크레딧 거래 원장
  payments      Payment[]       // 결제 이력
  subscriptions Subscription[]  // 구독 정보
  pptJobs       PptJob[]        // PPT 생성 작업
  
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}

enum Role {
  GENERAL      // 무료체험 3건 + 건당결제
  PRO          // 월 100건 무제한 (₩79,000/월)
  ENTERPRISE   // API 무제한 + 우선지원
}

// 크레딧 원장 (모든 거래 기록, 절대 삭제 금지)
model Credit {
  id          String   @id @default(cuid())
  userId      String
  user        User     @relation(fields: [userId], references: [id])
  
  amount      Int                // +3(가입), -1(PPT생성), +10(충전)
  balance     Int                // 거래 후 잔액 스냅샷
  type        CreditType         
  reason      String             // "회원가입 무료체험", "PPT 생성 차감", "Pro 구독 충전"
  
  relatedId   String?            // 연관 Payment/PptJob ID
  expiresAt   DateTime?          // 만료일 (Pro: 매월 1일 자동 리셋)
  
  createdAt   DateTime @default(now())
  
  @@index([userId, createdAt])  // 잔액 조회 최적화
}

enum CreditType {
  WELCOME_BONUS    // 가입 시 3건
  PURCHASE         // 건당 결제로 충전
  SUBSCRIPTION     // 구독으로 충전 (월 100건)
  USAGE            // PPT 생성 차감
  REFUND           // 생성 실패 시 환불
  EXPIRATION       // 만료 처리
}

// 결제 이력 (매출 분석용, 환불 추적)
model Payment {
  id              String   @id @default(cuid())
  userId          String
  user            User     @relation(fields: [userId], references: [id])
  
  orderId         String   @unique           // "order_20250127_abc123"
  pgProvider      String   @default("TOSS")  // TOSS/PORTONE
  pgTxId          String?  @unique           // PG사 거래 ID
  
  amount          Int                        // 결제 금액 (₩9,900 or ₩79,000)
  creditAmount    Int                        // 지급 크레딧 수 (1 or 100)
  paymentMethod   PaymentMethod              // CARD/ACCOUNT/PHONE
  
  status          PaymentStatus @default(PENDING)
  failureReason   String?
  
  webhookReceived DateTime?                  // 웹훅 수신 시각
  createdAt       DateTime @default(now())
  
  @@index([userId, status])
  @@index([createdAt])  // 일별 매출 집계
}

enum PaymentMethod {
  CARD           // 신용/체크카드 (2.3%)
  ACCOUNT        // 계좌이체 (0.7%, 1% 할인 쿠폰)
  PHONE          // 휴대폰 (5.5%, 비추천)
  TOSS_PAY       // 토스페이 간편결제 (2.3%)
}

enum PaymentStatus {
  PENDING        // 결제 요청 중
  SUCCESS        // 성공 (크레딧 자동 충전)
  FAILED         // 실패 (3회 재시도 후)
  REFUNDED       // 환불 완료
}

// 구독 정보
model Subscription {
  id              String   @id @default(cuid())
  userId          String
  user            User     @relation(fields: [userId], references: [id])
  
  plan            SubscriptionPlan
  status          SubscriptionStatus @default(ACTIVE)
  
  billingKey      String?  @unique    // 토스 자동결제 키
  nextBillingDate DateTime             // 다음 결제일
  canceledAt      DateTime?
  
  createdAt       DateTime @default(now())
  
  @@index([userId, status])
}

enum SubscriptionPlan {
  PRO            // ₩79,000/월, 월 100건
  ENTERPRISE     // ₩299,000/월, API 무제한
}

enum SubscriptionStatus {
  ACTIVE         // 정상 구독 중
  PAUSED         // 일시정지 (결제 실패)
  CANCELED       // 해지 (기간 만료 시 GENERAL 전환)
}

// PPT 생성 작업 (크레딧 차감과 트랜잭션 연동)
model PptJob {
  id          String   @id @default(cuid())
  userId      String
  user        User     @relation(fields: [userId], references: [id])
  
  naverUrl    String              // 입력 URL
  status      JobStatus @default(PENDING)
  resultUrl   String?             // S3 업로드 URL
  
  creditDeducted Boolean @default(false)  // 크레딧 차감 여부
  creditId    String?             // 차감 Credit.id (환불 추적용)
  
  errorMsg    String?
  createdAt   DateTime @default(now())
  completedAt DateTime?
  
  @@index([userId, status])
}

enum JobStatus {
  PENDING     // 큐 대기
  PROCESSING  // 크롤링 중
  SUCCESS     // 완료 (S3 업로드 완료)
  FAILED      // 실패 (크레딧 자동 환불)
}
```
